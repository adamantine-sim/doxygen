76 template <int dim, bool use_table, int p_order, int fe_degree,
77  typename MaterialStates, typename MemorySpaceType>
78 void ThermalOperator<dim, use_table, p_order, fe_degree, MaterialStates,
79  MemorySpaceType>::
80  cell_local_mass(
81  dealii::MatrixFree<dim, double> const &data,
82  dealii::LA::distributed::Vector<double, MemorySpaceType> &dst,
83  dealii::LA::distributed::Vector<double, MemorySpaceType> const &src,
84  std::pair<unsigned int, unsigned int> const &cell_range) const
85 {
86  // Get the subrange of cells associated with the fe index 0
87  std::pair<unsigned int, unsigned int> cell_subrange =
88  data.create_cell_subrange_hp_by_index(cell_range, 0);
89  dealii::FEEvaluation<dim, fe_degree, fe_degree + 1, 1, double> fe_eval(data);
90 
91  // Loop over the "cells". Note that we don't really work on a cell but on a
92  // set of quadrature point.
93  for (unsigned int cell = cell_subrange.first; cell < cell_subrange.second;
94  ++cell)
95  {
96  // Reinit fe_eval on the current cell
97  fe_eval.reinit(cell);
98  // Store in a local vector the local values of src
99  fe_eval.read_dof_values(src);
100  // Evaluate the shape function on the reference cell
101  fe_eval.evaluate(dealii::EvaluationFlags::values);
102  // Apply the Jacobian of the transformation
103  for (unsigned int q = 0; q < fe_eval.n_q_points; ++q)
104  {
105  fe_eval.submit_value(fe_eval.get_value(q), q);
106  }
107  // Sum over the quadrature points.
108  fe_eval.integrate(dealii::EvaluationFlags::values);
109  fe_eval.distribute_local_to_global(dst);
110  }
111 }

113 template <int dim, bool use_table, int p_order, int fe_degree,
114  typename MaterialStates, typename MemorySpaceType>
115 void ThermalOperator<dim, use_table, p_order, fe_degree, MaterialStates,
116  MemorySpaceType>::
117  compute_inverse_mass_matrix(
118  dealii::DoFHandler<dim> const &dof_handler,
119  dealii::AffineConstraints<double> const &affine_constraints)
120 {
121  // Compute the inverse of the mass matrix
122  dealii::hp::QCollection<dim> mass_q_collection;
123  mass_q_collection.push_back(dealii::QGaussLobatto<dim>(fe_degree + 1));
124  mass_q_collection.push_back(dealii::QGaussLobatto<dim>(2));
125 
126  typename dealii::MatrixFree<dim, double>::AdditionalData
127  mass_matrix_free_data;
128  mass_matrix_free_data.tasks_parallel_scheme =
129  dealii::MatrixFree<dim, double>::AdditionalData::partition_color;
130  mass_matrix_free_data.mapping_update_flags =
131  dealii::update_values | dealii::update_JxW_values;
132 
133  dealii::MatrixFree<dim, double> mass_matrix_free;
134  mass_matrix_free.reinit(dealii::StaticMappingQ1<dim>::mapping, dof_handler,
135  affine_constraints, mass_q_collection,
136  mass_matrix_free_data);
137  mass_matrix_free.initialize_dof_vector(*_inverse_mass_matrix);
138  dealii::LA::distributed::Vector<double, MemorySpaceType> unit_vector;
139  mass_matrix_free.initialize_dof_vector(unit_vector);
140  unit_vector = 1.;
141  mass_matrix_free.cell_loop(&ThermalOperator::cell_local_mass, this,
142  *_inverse_mass_matrix, unit_vector);
143  // Because cell_loop resolves the constraints, the constrained dofs are not
144  // called they stay at zero. Thus, we need to force the value on the
145  // constrained dofs by hand.
146  std::vector<unsigned int> const &constrained_dofs =
147  mass_matrix_free.get_constrained_dofs();
148  for (auto &dof : constrained_dofs)
149  _inverse_mass_matrix->local_element(dof) += 1.;
150 
151  _inverse_mass_matrix->compress(dealii::VectorOperation::add);
152 
153  unsigned int const local_size = _inverse_mass_matrix->locally_owned_size();
154  for (unsigned int k = 0; k < local_size; ++k)
155  {
156  _inverse_mass_matrix->local_element(k) =
157  1. / _inverse_mass_matrix->local_element(k);
158  }
159 }

195 template <int dim, bool use_table, int p_order, int fe_degree,
196  typename MaterialStates, typename MemorySpaceType>
197 void ThermalOperator<dim, use_table, p_order, fe_degree, MaterialStates,
198  MemorySpaceType>::
199  vmult_add(dealii::LA::distributed::Vector<double, MemorySpaceType> &dst,
200  dealii::LA::distributed::Vector<double, MemorySpaceType> const
201  &src) const
202 {
203  // Execute the matrix-free matrix-vector multiplication
204 
205  // If we use adiabatic boundary condition, we have nothing to do on the faces
206  // of the cell
207  if (_boundary_type & BoundaryType::adiabatic)
208  {
209  _matrix_free.cell_loop(&ThermalOperator::cell_local_apply, this, dst, src);
210  }
211  else
212  {
213  // MatrixFree::loop works like cell_loop but also allow computation on
214  // internal faces and boundary faces. Here, we use the same function for
215  // both cases and apply the face condition only at the boundary of the
216  // activated domain.
217  _matrix_free.loop(&ThermalOperator::cell_local_apply,
218  &ThermalOperator::face_local_apply,
219  &ThermalOperator::face_local_apply, this, dst, src);
220  }
221 
222  // Because cell_loop resolves the constraints, the constrained dofs are not
223  // called they stay at zero. Thus, we need to force the value on the
224  // constrained dofs by hand. The variable scaling is used so that we get the
225  // right order of magnitude.
226  // TODO: for now the value of scaling is set to 1
227  double const scaling = 1.;
228  std::vector<unsigned int> const &constrained_dofs =
229  _matrix_free.get_constrained_dofs();
230  for (auto &dof : constrained_dofs)
231  dst.local_element(dof) += scaling * src.local_element(dof);
232 }

246 template <int dim, bool use_table, int p_order, int fe_degree,
247  typename MaterialStates, typename MemorySpaceType>
248 void ThermalOperator<dim, use_table, p_order, fe_degree, MaterialStates,
249  MemorySpaceType>::
250  update_state_ratios(
251  [[maybe_unused]] unsigned int cell, [[maybe_unused]] unsigned int q,
252  [[maybe_unused]] dealii::VectorizedArray<double> temperature,
253  std::array<dealii::VectorizedArray<double>,
254  MaterialStates::n_material_states> &state_ratios) const
255 {
256  unsigned int constexpr solid =
257  static_cast<unsigned int>(MaterialStates::State::solid);
258  if constexpr (std::is_same_v<MaterialStates, Solid>)
259  {
260  state_ratios[solid] = 1.;
261  }
262  else if constexpr (std::is_same_v<MaterialStates, SolidLiquid>)
263  {
264  unsigned int constexpr liquid =
265  static_cast<unsigned int>(MaterialStates::State::liquid);
266  // Loop over the vectorized arrays
267  for (unsigned int n = 0; n < temperature.size(); ++n)
268  {
269  // Get the material id at this point
270  dealii::types::material_id const material_id = _material_id(cell, q)[n];
271 
272  // Get the material thermodynamic properties
273  double const solidus =
274  _material_properties.get(material_id, Property::solidus);
275  double const liquidus =
276  _material_properties.get(material_id, Property::liquidus);
277 
278  // Update the state ratios
279  if (temperature[n] < solidus)
280  state_ratios[liquid][n] = 0.;
281  else if (temperature[n] > liquidus)
282  state_ratios[liquid][n] = 1.;
283  else
284  {
285  state_ratios[liquid][n] =
286  (temperature[n] - solidus) / (liquidus - solidus);
287  }
288  state_ratios[solid][n] = 1. - state_ratios[liquid][n];
289  }
290 
291  _liquid_ratio(cell, q) = state_ratios[liquid];
292  }
293  else if constexpr (std::is_same_v<MaterialStates, SolidLiquidPowder>)
294  {
295  unsigned int constexpr liquid =
296  static_cast<unsigned int>(MaterialStates::State::liquid);
297  unsigned int constexpr powder =
298  static_cast<unsigned int>(MaterialStates::State::powder);
299 
300  // Loop over the vectorized arrays
301  for (unsigned int n = 0; n < temperature.size(); ++n)
302  {
303  // Get the material id at this point
304  dealii::types::material_id const material_id = _material_id(cell, q)[n];
305 
306  // Get the material thermodynamic properties
307  double const solidus =
308  _material_properties.get(material_id, Property::solidus);
309  double const liquidus =
310  _material_properties.get(material_id, Property::liquidus);
311 
312  // Update the state ratios
313  state_ratios[powder] = _powder_ratio(cell, q);
314 
315  if (temperature[n] < solidus)
316  state_ratios[liquid][n] = 0.;
317  else if (temperature[n] > liquidus)
318  state_ratios[liquid][n] = 1.;
319  else
320  {
321  state_ratios[liquid][n] =
322  (temperature[n] - solidus) / (liquidus - solidus);
323  }
324  // Because the powder can only become liquid, the solid can only
325  // become liquid, and the liquid can only become solid, the ratio of
326  // powder can only decrease.
327  state_ratios[powder][n] =
328  std::min(1. - state_ratios[liquid][n], state_ratios[powder][n]);
329  state_ratios[solid][n] =
330  1. - state_ratios[liquid][n] - state_ratios[powder][n];
331  }
332 
333  _liquid_ratio(cell, q) = state_ratios[liquid];
334  _powder_ratio(cell, q) = state_ratios[powder];
335  }
336 }

434 template <int dim, bool use_table, int p_order, int fe_degree,
435  typename MaterialStates, typename MemorySpaceType>
436 dealii::VectorizedArray<double>
437 ThermalOperator<dim, use_table, p_order, fe_degree, MaterialStates,
438  MemorySpaceType>::
439  get_inv_rho_cp(
440  std::array<dealii::types::material_id,
441  dealii::VectorizedArray<double>::size()> const &material_id,
442  std::array<dealii::VectorizedArray<double>,
443  MaterialStates::n_material_states> const &state_ratios,
444  dealii::VectorizedArray<double> const &temperature,
445  dealii::AlignedVector<dealii::VectorizedArray<double>> const
446  &temperature_powers) const
447 {
448  // Here we need the specific heat (including the latent heat contribution)
449  // and the density
450 
451  // Compute the state-dependent properties
452  dealii::VectorizedArray<double> density =
453  _material_properties.template compute_material_property<use_table>(
454  StateProperty::density, material_id.data(), state_ratios.data(),
455  temperature, temperature_powers);
456 
457  dealii::VectorizedArray<double> specific_heat =
458  _material_properties.template compute_material_property<use_table>(
459  StateProperty::specific_heat, material_id.data(), state_ratios.data(),
460  temperature, temperature_powers);
461 
462  // Add in the latent heat contribution
463  if constexpr (!std::is_same_v<MaterialStates, Solid>)
464  {
465  // Get the state-independent material properties
466  dealii::VectorizedArray<double> solidus, liquidus, latent_heat;
467  for (unsigned int n = 0; n < solidus.size(); ++n)
468  {
469  solidus[n] = _material_properties.get(material_id[n], Property::solidus);
470  liquidus[n] =
471  _material_properties.get(material_id[n], Property::liquidus);
472  latent_heat[n] =
473  _material_properties.get(material_id[n], Property::latent_heat);
474  }
475 
476  unsigned int constexpr solid =
477  static_cast<unsigned int>(MaterialStates::State::solid);
478  unsigned int constexpr liquid =
479  static_cast<unsigned int>(MaterialStates::State::liquid);
480 
481  // We only need to take the latent heat into account if both the liquid and
482  // the solid phases are present. We could use an if with two conditions but
483  // that is very slow. Instead, we create a new variable is_mushy that is
484  // non-zero when there is both solid and liquid.
485  auto is_mushy = state_ratios[liquid] * state_ratios[solid];
486  for (unsigned int n = 0; n < specific_heat.size(); ++n)
487  {
488  if (is_mushy[n] > 0.0)
489  {
490  specific_heat[n] += latent_heat[n] / (liquidus[n] - solidus[n]);
491  }
492  }
493  }
494 
495  return 1.0 / (density * specific_heat);
496 }

498 template <int dim, bool use_table, int p_order, int fe_degree,
499  typename MaterialStates, typename MemorySpaceType>
500 void ThermalOperator<dim, use_table, p_order, fe_degree, MaterialStates,
501  MemorySpaceType>::
502  cell_local_apply(
503  dealii::MatrixFree<dim, double> const &data,
504  dealii::LA::distributed::Vector<double, MemorySpaceType> &dst,
505  dealii::LA::distributed::Vector<double, MemorySpaceType> const &src,
506  std::pair<unsigned int, unsigned int> const &cell_range) const
507 {
508  // Get the subrange of cells associated with the fe index 0
509  std::pair<unsigned int, unsigned int> cell_subrange =
510  data.create_cell_subrange_hp_by_index(cell_range, 0);
511 
512  dealii::FEEvaluation<dim, fe_degree, fe_degree + 1, 1, double> fe_eval(data);
513  std::array<dealii::VectorizedArray<double>, MaterialStates::n_material_states>
514  state_ratios;
515 
516  // We need powers of temperature to compute the material properties. We
517  // could compute it in MaterialProperty but because it's in a hot loop.
518  // It's really worth to compute it once and pass it when we compute a
519  // material property.
520  dealii::AlignedVector<dealii::VectorizedArray<double>> temperature_powers(
521  p_order + 1);
522 
523  // Loop over the "cells". Note that we don't really work on a cell but on a
524  // set of quadrature point.
525  for (unsigned int cell = cell_subrange.first; cell < cell_subrange.second;
526  ++cell)
527  {
528  // Reinit fe_eval on the current cell
529  fe_eval.reinit(cell);
530  // Store in a local vector the local values of src
531  fe_eval.read_dof_values(src);
532  // Evaluate the function and its gradient on the reference cell
533  fe_eval.evaluate(dealii::EvaluationFlags::values |
534  dealii::EvaluationFlags::gradients);
535  // Apply the Jacobian of the transformation, multiply by the variable
536  // coefficients and the quadrature points
537  for (unsigned int q = 0; q < fe_eval.n_q_points; ++q)
538  {
539  auto temperature = fe_eval.get_value(q);
540  // Precompute the powers of temperature.
541  for (unsigned int i = 0; i <= p_order; ++i)
542  {
543  // FIXME Need to cast i to double due to a limitation in deal.II 9.5
544  temperature_powers[i] = std::pow(temperature, static_cast<double>(i));
545  }
546 
547  // Calculate the local material properties
548  update_state_ratios(cell, q, temperature, state_ratios);
549  auto material_id = _material_id(cell, q);
550  auto inv_rho_cp = get_inv_rho_cp(material_id, state_ratios, temperature,
551  temperature_powers);
552  auto th_conductivity_grad = fe_eval.get_gradient(q);
553 
554  // In 2D we only use x and z, and there are no deposition angle
555  if constexpr (dim == 2)
556  {
557  th_conductivity_grad[axis<dim>::x] *=
558  _material_properties.template compute_material_property<use_table>(
559  StateProperty::thermal_conductivity_x, material_id.data(),
560  state_ratios.data(), temperature, temperature_powers);
561  th_conductivity_grad[axis<dim>::z] *=
562  _material_properties.template compute_material_property<use_table>(
563  StateProperty::thermal_conductivity_z, material_id.data(),
564  state_ratios.data(), temperature, temperature_powers);
565  }
566 
567  if constexpr (dim == 3)
568  {
569  auto const th_conductivity_grad_x = th_conductivity_grad[axis<dim>::x];
570  auto const th_conductivity_grad_y = th_conductivity_grad[axis<dim>::y];
571  auto const thermal_conductivity_x =
572  _material_properties.template compute_material_property<use_table>(
573  StateProperty::thermal_conductivity_x, material_id.data(),
574  state_ratios.data(), temperature, temperature_powers);
575  auto const thermal_conductivity_y =
576  _material_properties.template compute_material_property<use_table>(
577  StateProperty::thermal_conductivity_y, material_id.data(),
578  state_ratios.data(), temperature, temperature_powers);
579 
580  auto cos = _deposition_cos(cell, q);
581  auto sin = _deposition_sin(cell, q);
582 
583  // The rotation is performed using the following formula
584  //
585  // (cos -sin) (x 0) ( cos sin)
586  // (sin cos) (0 y) (-sin cos)
587  // =
588  // ((x*cos^2 + y*sin^2) ((x-y) * (sin*cos)))
589  // (((x-y) * (sin*cos)) (x*sin^2 + y*cos^2))
590 
591  th_conductivity_grad[axis<dim>::x] =
592  (thermal_conductivity_x * cos * cos +
593  thermal_conductivity_y * sin * sin) *
594  th_conductivity_grad_x +
595  ((thermal_conductivity_x - thermal_conductivity_y) * sin * cos) *
596  th_conductivity_grad_y;
597  th_conductivity_grad[axis<dim>::y] =
598  ((thermal_conductivity_x - thermal_conductivity_y) * sin * cos) *
599  th_conductivity_grad_x +
600  (thermal_conductivity_x * sin * sin +
601  thermal_conductivity_y * cos * cos) *
602  th_conductivity_grad_y;
603 
604  // There is no deposition angle for the z axis
605  th_conductivity_grad[axis<dim>::z] *=
606  _material_properties.template compute_material_property<use_table>(
607  StateProperty::thermal_conductivity_z, material_id.data(),
608  state_ratios.data(), temperature, temperature_powers);
609  }
610 
611  fe_eval.submit_gradient(-inv_rho_cp * th_conductivity_grad, q);
612 
613  // Compute source term
614  dealii::Point<dim, dealii::VectorizedArray<double>> const &q_point =
615  fe_eval.quadrature_point(q);
616 
617  dealii::VectorizedArray<double> quad_pt_source = 0.0;
618  for (unsigned int i = 0;
619  i < _matrix_free.n_active_entries_per_cell_batch(cell); ++i)
620  {
621  dealii::Point<dim> q_point_loc;
622  for (unsigned int d = 0; d < dim; ++d)
623  q_point_loc(d) = q_point(d)[i];
624 
625  for (auto &beam : _heat_sources)
626  quad_pt_source[i] += beam->value(q_point_loc, _current_source_height);
627  }
628  quad_pt_source *= inv_rho_cp;
629 
630  fe_eval.submit_value(quad_pt_source, q);
631  }
632  // Sum over the quadrature points.
633  fe_eval.integrate(dealii::EvaluationFlags::values |
634  dealii::EvaluationFlags::gradients);
635  fe_eval.distribute_local_to_global(dst);
636  }
637 }

639 template <int dim, bool use_table, int p_order, int fe_degree,
640  typename MaterialStates, typename MemorySpaceType>
641 void ThermalOperator<dim, use_table, p_order, fe_degree, MaterialStates,
642  MemorySpaceType>::
643  face_local_apply(
644  dealii::MatrixFree<dim, double> const &data,
645  dealii::LA::distributed::Vector<double, MemorySpaceType> &dst,
646  dealii::LA::distributed::Vector<double, MemorySpaceType> const &src,
647  std::pair<unsigned int, unsigned int> const &face_range) const
648 {
649  // Get the fe_indices of the cells that share faces in face_range;
650  auto const adjacent_cells_fe_index = data.get_face_range_category(face_range);
651  // We now have four cases:
652  // - cell_1 = cell_2 = FE_Q: internal face of the activated domain
653  // - cell_1/2 = FE_Q and cell_2/1 = FE_Nothing/does not exit: boundary of
654  // the
655  // activated domain
656  // - cell_1/2 = FE_Nothing and cell_2/1 = does not exit: external boundary
657  // of
658  // the deactivated domain
659  // - cell_1 = cell_2 = FE_Nothing: internal face of the non-activated
660  // domain
661  // Since we only care on the faces that are at the boundary of the activated
662  // domain, we need to check that cell_1 is different than cell_2 and that at
663  // one of the two cells is using FE_Q
664  if (adjacent_cells_fe_index.first == adjacent_cells_fe_index.second)
665  {
666  return;
667  }
668  if ((adjacent_cells_fe_index.first != 0 &&
669  adjacent_cells_fe_index.second != 0))
670  {
671  return;
672  }
673 
674  // Create the FEFaceEvaluation object. The boolean in the constructor is
675  // used to decided which cell the face should be exterior to.
676  dealii::FEFaceEvaluation<dim, fe_degree, fe_degree + 1, 1, double>
677  fe_face_eval(data, adjacent_cells_fe_index.first == 0);
678  std::array<dealii::VectorizedArray<double>, MaterialStates::n_material_states>
679  face_state_ratios;
680 
681  // Create variables used to compute boundary conditions.
682  auto conv_temperature_infty = dealii::make_vectorized_array<double>(0.);
683  auto conv_heat_transfer_coef = dealii::make_vectorized_array<double>(0.);
684  auto rad_temperature_infty = dealii::make_vectorized_array<double>(0.);
685  auto rad_heat_transfer_coef = dealii::make_vectorized_array<double>(0.);
686 
687  // We need powers of temperature to compute the material properties. We
688  // could compute it in MaterialProperty but because it's in a hot loop.
689  // It's really worth to compute it once and pass it when we compute a
690  // material property.
691  dealii::AlignedVector<dealii::VectorizedArray<double>> temperature_powers(
692  p_order + 1);
693 
694  // Loop over the faces
695  for (unsigned int face = face_range.first; face < face_range.second; ++face)
696  {
697  // Reinit fe_face_eval on the current face
698  fe_face_eval.reinit(face);
699  // Store in a local vector the local values of src
700  fe_face_eval.read_dof_values(src);
701  // Evalue the function on the reference cell
702  fe_face_eval.evaluate(dealii::EvaluationFlags::values);
703  // Apply the Jacobian of the transformation, mutliply by the variable
704  // coefficients and the quadrature points
705  for (unsigned int q = 0; q < fe_face_eval.n_q_points; ++q)
706  {
707  auto temperature = fe_face_eval.get_value(q);
708  // Precompute the powers of temperature.
709  for (unsigned int i = 0; i <= p_order; ++i)
710  {
711  // FIXME Need to cast i to double due to a limitation in deal.II 9.5
712  temperature_powers[i] = std::pow(temperature, static_cast<double>(i));
713  }
714 
715  // Compute the local_properties
716  auto material_id = _face_material_id(face, q);
717  update_face_state_ratios(face, q, temperature, face_state_ratios);
718  auto const inv_rho_cp = get_inv_rho_cp(material_id, face_state_ratios,
719  temperature, temperature_powers);
720  if (_boundary_type & BoundaryType::convective)
721  {
722  for (unsigned int n = 0; n < conv_temperature_infty.size(); ++n)
723  {
724  conv_temperature_infty[n] = _material_properties.get(
725  material_id[n], Property::convection_temperature_infty);
726  }
727  conv_heat_transfer_coef =
728  _material_properties.template compute_material_property<use_table>(
729  StateProperty::convection_heat_transfer_coef,
730  material_id.data(), face_state_ratios.data(), temperature,
731  temperature_powers);
732  }
733  if (_boundary_type & BoundaryType::radiative)
734  {
735  for (unsigned int n = 0; n < rad_temperature_infty.size(); ++n)
736  {
737  rad_temperature_infty[n] = _material_properties.get(
738  material_id[n], Property::radiation_temperature_infty);
739  }
740 
741  // We need the radiation heat transfer coefficient but it is not a
742  // real material property but it is derived from other material
743  // properties: h_rad = emissitivity * stefan-boltzmann constant * (T
744  // + T_infty) (T^2 + T^2_infty).
745  rad_heat_transfer_coef =
746  _material_properties.template compute_material_property<use_table>(
747  StateProperty::emissivity, material_id.data(),
748  face_state_ratios.data(), temperature, temperature_powers) *
749  Constant::stefan_boltzmann * (temperature + rad_temperature_infty) *
750  (temperature * temperature +
751  rad_temperature_infty * rad_temperature_infty);
752  }
753 
754  auto const boundary_val =
755  -inv_rho_cp *
756  (conv_heat_transfer_coef * (temperature - conv_temperature_infty) +
757  rad_heat_transfer_coef * (temperature - rad_temperature_infty));
758  fe_face_eval.submit_value(boundary_val * fe_face_eval.get_value(q), q);
759  }
760  // Sum over the quadrature points
761  fe_face_eval.integrate(dealii::EvaluationFlags::values);
762  fe_face_eval.distribute_local_to_global(dst);
763  }
764 }
