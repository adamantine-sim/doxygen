192 template <int dim>
193 void StrainPostProcessor<dim>::evaluate_vector_field(
194  const dealii::DataPostprocessorInputs::Vector<dim> &displacement_data,
195  std::vector<dealii::Vector<double>> &strain) const
196 {
197  for (unsigned int i = 0; i < displacement_data.solution_gradients.size(); ++i)
198  {
199  for (unsigned int j = 0; j < dim; ++j)
200  {
201  for (unsigned int k = 0; k < dim; ++k)
202  {
203  // strain = (\nabla u + (\nabla u)^T)/2
204  strain[i][dealii::Tensor<2, dim>::component_to_unrolled_index(
205  dealii::TableIndices<2>(j, k))] =
206  (displacement_data.solution_gradients[i][j][k] +
207  displacement_data.solution_gradients[i][k][j]) /
208  2.;
209  }
210  }
211  }
212 }

279 template <int dim>
280 template <typename LayoutType>
281 void PostProcessor<dim>::material_dataout(
282  Kokkos::View<double **, LayoutType, kokkos_host> state,
283  std::unordered_map<dealii::types::global_dof_index, unsigned int> const
284  &dofs_map,
285  dealii::DoFHandler<dim> const &material_dof_handler)
286 {
287  unsigned int const n_active_cells =
288  material_dof_handler.get_triangulation().n_active_cells();
289  dealii::Vector<double> powder(n_active_cells);
290  dealii::Vector<double> liquid(n_active_cells);
291  dealii::Vector<double> solid(n_active_cells);
292  unsigned int constexpr powder_index =
293  static_cast<unsigned int>(SolidLiquidPowder::State::powder);
294  unsigned int constexpr liquid_index =
295  static_cast<unsigned int>(SolidLiquidPowder::State::liquid);
296  unsigned int constexpr solid_index =
297  static_cast<unsigned int>(SolidLiquidPowder::State::solid);
298  auto mp_cell = material_dof_handler.begin_active();
299  auto mp_end_cell = material_dof_handler.end();
300  std::vector<dealii::types::global_dof_index> mp_dof_indices(1);
301  for (unsigned int i = 0; mp_cell != mp_end_cell; ++i, ++mp_cell)
302  if (mp_cell->is_locally_owned())
303  {
304  mp_cell->get_dof_indices(mp_dof_indices);
305  dealii::types::global_dof_index const mp_dof_index =
306  dofs_map.at(mp_dof_indices[0]);
307  solid[i] = state(solid_index, mp_dof_index);
308  liquid[i] = liquid_index < state.extent(0)
309  ? state(liquid_index, mp_dof_index)
310  : 0.;
311  powder[i] = powder_index < state.extent(0)
312  ? state(powder_index, mp_dof_index)
313  : 0;
314  }
315  _data_out.add_data_vector(powder, "powder");
316  _data_out.add_data_vector(liquid, "liquid");
317  _data_out.add_data_vector(solid, "solid");
318 }
