417 template <int dim, int p_order, typename MaterialStates,
418  typename MemorySpaceType>
419 template <bool use_table>
420 dealii::VectorizedArray<double>
421 MaterialProperty<dim, p_order, MaterialStates, MemorySpaceType>::
422  compute_material_property(
423  StateProperty state_property,
424  dealii::types::material_id const *material_id,
425  dealii::VectorizedArray<double> const *state_ratios,
426  dealii::VectorizedArray<double> const &temperature,
427  dealii::AlignedVector<dealii::VectorizedArray<double>> const
428  &temperature_powers) const
429 {
430  dealii::VectorizedArray<double> value = 0.0;
431  unsigned int const property_index = static_cast<unsigned int>(state_property);
432 
433  if constexpr (use_table)
434  {
435  for (unsigned int material_state = 0;
436  material_state < MaterialStates::n_material_states; ++material_state)
437  {
438  for (unsigned int n = 0; n < dealii::VectorizedArray<double>::size(); ++n)
439  {
440  const dealii::types::material_id m_id = material_id[n];
441 
442  value[n] += state_ratios[material_state][n] *
443  compute_property_from_table(_state_property_tables, m_id,
444  material_state, property_index,
445  temperature[n]);
446  }
447  }
448  }
449  else
450  {
451  for (unsigned int material_state = 0;
452  material_state < MaterialStates::n_material_states; ++material_state)
453  {
454  for (unsigned int n = 0; n < dealii::VectorizedArray<double>::size(); ++n)
455  {
456  dealii::types::material_id m_id = material_id[n];
457 
458  for (unsigned int i = 0; i <= p_order; ++i)
459  {
460  value[n] += state_ratios[material_state][n] *
461  _state_property_polynomials(m_id, material_state,
462  property_index, i) *
463  temperature_powers[i][n];
464  }
465  }
466  }
467  }
468 
469  return value;
470 }

472 template <int dim, int p_order, typename MaterialStates,
473  typename MemorySpaceType>
474 template <bool use_table>
475 KOKKOS_FUNCTION double
476 MaterialProperty<dim, p_order, MaterialStates, MemorySpaceType>::
477  compute_material_property(StateProperty state_property,
478  dealii::types::material_id const material_id,
479  double const *state_ratios,
480  double temperature) const
481 {
482  double value = 0.0;
483  unsigned int const property_index = static_cast<unsigned int>(state_property);
484 
485  if constexpr (use_table)
486  {
487  for (unsigned int material_state = 0;
488  material_state < MaterialStates::n_material_states; ++material_state)
489  {
490  const dealii::types::material_id m_id = material_id;
491 
492  value += state_ratios[material_state] *
493  compute_property_from_table(_state_property_tables, m_id,
494  material_state, property_index,
495  temperature);
496  }
497  }
498  else
499  {
500  for (unsigned int material_state = 0;
501  material_state < MaterialStates::n_material_states; ++material_state)
502  {
503  dealii::types::material_id m_id = material_id;
504 
505  for (unsigned int i = 0; i <= p_order; ++i)
506  {
507  value += state_ratios[material_state] *
508  _state_property_polynomials(m_id, material_state,
509  property_index, i) *
510  std::pow(temperature, i);
511  }
512  }
513  }
514 
515  return value;
