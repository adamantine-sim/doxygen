21 template <int dim, int p_order, typename MaterialStates,
22  typename MemorySpaceType>
23 MechanicalPhysics<dim, p_order, MaterialStates, MemorySpaceType>::
24  MechanicalPhysics(MPI_Comm const &communicator,
25  unsigned int const fe_degree, Geometry<dim> &geometry,
26  MaterialProperty<dim, p_order, MaterialStates,
27  MemorySpaceType> &material_properties,
28  std::vector<double> const &reference_temperatures)
29  : _geometry(geometry), _material_properties(material_properties),
30  _dof_handler(_geometry.get_triangulation()),
31  _solution_transfer(_dof_handler),
32  _cell_data_transfer(
33  dynamic_cast<const dealii::parallel::distributed::Triangulation<dim>
34  &>(_dof_handler.get_triangulation()))
35 {
36  // Create the FECollection
37  _fe_collection.push_back(
38  dealii::FESystem<dim>(dealii::FE_Q<dim>(fe_degree) ^ dim));
39  _fe_collection.push_back(
40  dealii::FESystem<dim>(dealii::FE_Nothing<dim>() ^ dim));
41 
42  // Create the QCollection
43  _q_collection.push_back(dealii::QGauss<dim>(fe_degree + 1));
44  _q_collection.push_back(dealii::QGauss<dim>(1));
45 
46  // Solve the mechanical problem only on the part of the domain that has solid
47  // material.
48  unsigned int n_active_cells =
49  _dof_handler.get_triangulation().n_active_cells();
50  for (auto const &cell :
51  dealii::filter_iterators(_dof_handler.active_cell_iterators(),
52  dealii::IteratorFilters::LocallyOwnedCell()))
53  {
54  if (_material_properties.get_state_ratio(
55  cell, MaterialStates::State::solid) > 0.99)
56  {
57  cell->set_active_fe_index(0);
58  }
59  else
60  {
61  cell->set_active_fe_index(1);
62  }
63  }
64 
65  // Create the mechanical operator
66  _mechanical_operator = std::make_unique<
67  MechanicalOperator<dim, p_order, MaterialStates, MemorySpaceType>>(
68  communicator, _material_properties, reference_temperatures);
69 
70  // Create the data used to compute the stress tensor
71  unsigned int const n_quad_pts = _q_collection.max_n_quadrature_points();
72  _plastic_internal_variable.reserve(n_active_cells);
73 
74  for (auto const &cell : _dof_handler.active_cell_iterators())
75  {
76  if (cell->is_locally_owned())
77  {
78  auto elastic_limit = _material_properties.get_mechanical_property(
79  cell, StateProperty::elastic_limit);
80  _plastic_internal_variable.emplace_back(
81  std::vector<double>(n_quad_pts, elastic_limit));
82  }
83  else
84  {
85  _plastic_internal_variable.emplace_back(std::vector<double>(
86  n_quad_pts, std::numeric_limits<double>::signaling_NaN()));
87  }
88  }
89  _stress.resize(n_active_cells,
90  std::vector<dealii::SymmetricTensor<2, dim>>(n_quad_pts));
91  _back_stress.resize(n_active_cells,
92  std::vector<dealii::SymmetricTensor<2, dim>>(n_quad_pts));
93 }

95 template <int dim, int p_order, typename MaterialStates,
96  typename MemorySpaceType>
97 void MechanicalPhysics<dim, p_order, MaterialStates, MemorySpaceType>::
98  setup_dofs(std::vector<std::shared_ptr<BodyForce<dim>>> const &body_forces)
99 {
100  _dof_handler.distribute_dofs(_fe_collection);
101  dealii::IndexSet locally_relevant_dofs;
102  dealii::DoFTools::extract_locally_relevant_dofs(_dof_handler,
103  locally_relevant_dofs);
104  _affine_constraints.clear();
105  _affine_constraints.reinit(locally_relevant_dofs);
106  dealii::DoFTools::make_hanging_node_constraints(_dof_handler,
107  _affine_constraints);
108  // FIXME For now this is only a Dirichlet boundary condition. It is also
109  // manually set to be what is the bottom face for a dealii hyper-rectangle. We
110  // need to decide how we want to expose BC control to the user more generally
111  // (including for user-supplied meshes).
112  dealii::VectorTools::interpolate_boundary_values(
113  _dof_handler, 4, dealii::Functions::ZeroFunction<dim>(dim),
114  _affine_constraints);
115  _affine_constraints.close();
116 
117  _mechanical_operator->reinit(_dof_handler, _affine_constraints, _q_collection,
118  body_forces);
119 }

121 template <int dim, int p_order, typename MaterialStates,
122  typename MemorySpaceType>
123 void MechanicalPhysics<dim, p_order, MaterialStates,
124  MemorySpaceType>::prepare_transfer_mpi()
125 {
126  _old_displacement.update_ghost_values();
127  _solution_transfer.prepare_for_coarsening_and_refinement(_old_displacement);
128 
129  _data_to_transfer.clear();
130  unsigned int const n_quad_pts = _q_collection.max_n_quadrature_points();
131  unsigned int const n_doubles_per_quad_plastic = 1;
132  unsigned int const n_doubles_per_quad_stress =
133  dealii::SymmetricTensor<2, dim>::n_independent_components;
134 
135  unsigned int const n_doubles_per_quad =
136  n_doubles_per_quad_plastic + n_doubles_per_quad_stress * 2;
137  std::vector<double> dummy_cell_data(n_quad_pts * n_doubles_per_quad,
138  std::numeric_limits<double>::infinity());
139 
140  unsigned int cell_id = 0;
141  for (auto const &cell : _dof_handler.active_cell_iterators())
142  {
143  if (cell->is_locally_owned())
144  {
145  std::vector<double> cell_data(n_quad_pts * n_doubles_per_quad);
146  unsigned int const stress_offset =
147  n_quad_pts * n_doubles_per_quad_plastic;
148  unsigned int const back_stress_offset =
149  n_quad_pts * (n_doubles_per_quad_plastic + n_doubles_per_quad_stress);
150 
151  std::copy(_plastic_internal_variable[cell_id].begin(),
152  _plastic_internal_variable[cell_id].end(), cell_data.begin());
153 
154  for (unsigned int quad = 0; quad < n_quad_pts; ++quad)
155  {
156  for (unsigned int i = 0; i < n_doubles_per_quad_stress; ++i)
157  {
158  cell_data[stress_offset + quad * n_doubles_per_quad_stress + i] =
159  _stress[cell_id][quad].access_raw_entry(i);
160  cell_data[back_stress_offset + quad * n_doubles_per_quad_stress + i] =
161  _back_stress[cell_id][quad].access_raw_entry(i);
162  }
163  }
164  _data_to_transfer.push_back(std::move(cell_data));
165  }
166  else
167  {
168  _data_to_transfer.push_back(dummy_cell_data);
169  }
170  ++cell_id;
171  }
172  _cell_data_transfer.prepare_for_coarsening_and_refinement(_data_to_transfer);
173 }

175 template <int dim, int p_order, typename MaterialStates,
176  typename MemorySpaceType>
177 void MechanicalPhysics<dim, p_order, MaterialStates,
178  MemorySpaceType>::complete_transfer_mpi()
179 {
180  _dof_handler.distribute_dofs(_fe_collection);
181 
182  const dealii::IndexSet locally_relevant_dofs =
183  dealii::DoFTools::extract_locally_relevant_dofs(_dof_handler);
184  _old_displacement.reinit(_dof_handler.locally_owned_dofs(),
185  locally_relevant_dofs,
186  _dof_handler.get_communicator());
187  _solution_transfer.interpolate(_old_displacement);
188 
189  auto n_active_cells = _dof_handler.get_triangulation().n_active_cells();
190  unsigned int const n_quad_pts = _q_collection.max_n_quadrature_points();
191 
192  _plastic_internal_variable.resize(n_active_cells,
193  std::vector<double>(n_quad_pts));
194  _stress.resize(n_active_cells,
195  std::vector<dealii::SymmetricTensor<2, dim>>(n_quad_pts));
196  _back_stress.resize(n_active_cells,
197  std::vector<dealii::SymmetricTensor<2, dim>>(n_quad_pts));
198 
199  unsigned int const n_doubles_per_quad_plastic = 1;
200  unsigned int const n_doubles_per_quad_stress =
201  dealii::SymmetricTensor<2, dim>::n_independent_components;
202 
203  unsigned int const n_doubles_per_quad =
204  n_doubles_per_quad_plastic + n_doubles_per_quad_stress * 2;
205  std::vector<std::vector<double>> data_to_unpack(
206  n_active_cells, std::vector<double>(n_doubles_per_quad * n_quad_pts));
207  _cell_data_transfer.unpack(data_to_unpack);
208 
209  unsigned int cell_id = 0;
210  for (auto const &cell : _dof_handler.active_cell_iterators())
211  {
212  if (cell->is_locally_owned())
213  {
214  unsigned int const stress_offset =
215  n_quad_pts * n_doubles_per_quad_plastic;
216  unsigned int const back_stress_offset =
217  n_quad_pts * (n_doubles_per_quad_plastic + n_doubles_per_quad_stress);
218 
219  std::copy(data_to_unpack[cell_id].begin(),
220  data_to_unpack[cell_id].begin() + stress_offset,
221  _plastic_internal_variable[cell_id].begin());
222 
223  for (unsigned int quad = 0; quad < n_quad_pts; ++quad)
224  {
225  for (unsigned int i = 0; i < n_doubles_per_quad_stress; ++i)
226  {
227  _stress[cell_id][quad].access_raw_entry(i) =
228  data_to_unpack[cell_id][stress_offset +
229  quad * n_doubles_per_quad_stress + i];
230  _back_stress[cell_id][quad].access_raw_entry(i) =
231  data_to_unpack[cell_id][back_stress_offset +
232  quad * n_doubles_per_quad_stress + i];
233  }
234  }
235  }
236  ++cell_id;
237  }
238 }

240 template <int dim, int p_order, typename MaterialStates,
241  typename MemorySpaceType>
242 void MechanicalPhysics<dim, p_order, MaterialStates, MemorySpaceType>::
243  setup_dofs(
244  dealii::DoFHandler<dim> const &thermal_dof_handler,
245  dealii::LA::distributed::Vector<double, dealii::MemorySpace::Host> const
246  &temperature,
247  std::vector<bool> const &has_melted,
248  std::vector<std::shared_ptr<BodyForce<dim>>> const &body_forces)
249 {
250  _mechanical_operator->update_temperature(thermal_dof_handler, temperature,
251  has_melted);
252  // Update the active fe indices, the plastic variables, and the displacement.
253  unsigned int const n_quad_pts = _q_collection.max_n_quadrature_points();
254  unsigned int cell_id = 0;
255  std::vector<std::vector<double>> saved_old_displacement;
256  std::vector<std::vector<double>> tmp_plastic_internal_variable;
257  std::vector<std::vector<dealii::SymmetricTensor<2, dim>>> tmp_stress;
258  std::vector<std::vector<dealii::SymmetricTensor<2, dim>>> tmp_back_stress;
259  // The number of cells to activate/deactive should be small, so we can
260  // already reserve the memory.
261  unsigned int const n_dofs_per_cell = _fe_collection.max_dofs_per_cell();
262  unsigned int const n_old_active_cells = _plastic_internal_variable.size();
263  std::vector<dealii::types::global_dof_index> global_dof_indices(
264  n_dofs_per_cell);
265  tmp_plastic_internal_variable.reserve(n_old_active_cells);
266  tmp_stress.reserve(n_old_active_cells);
267  tmp_back_stress.reserve(_back_stress.size());
268  // First we save _old_displacement if it exists
269  if (_old_displacement.size())
270  {
271  _old_displacement.update_ghost_values();
272 
273  std::vector<double> cell_values(n_dofs_per_cell);
274  saved_old_displacement.reserve(n_old_active_cells);
275  for (auto const &cell : _dof_handler.active_cell_iterators())
276  {
277  if (cell->is_locally_owned())
278  {
279  auto fe_index = cell->active_fe_index();
280  if (fe_index == 0)
281  {
282  // The cell contains solid material, we need to save the displacement
283  cell->get_dof_indices(global_dof_indices);
284  for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
285  {
286  cell_values[i] = _old_displacement[global_dof_indices[i]];
287  }
288  }
289  else
290  {
291  // The cell does not contain material or it is liquid. The
292  // displacement is ignored.
293  cell_values.assign(n_dofs_per_cell, 0.);
294  }
295  saved_old_displacement.push_back(cell_values);
296  }
297  else
298  {
299  saved_old_displacement.push_back(std::vector<double>(n_dofs_per_cell));
300  }
301  }
302  }
303 
304  // Now we can update the fe indices and the plastic variables.
305  for (auto const &cell : _dof_handler.active_cell_iterators())
306  {
307  if (cell->is_locally_owned())
308  {
309  auto current_fe_index = cell->active_fe_index();
310  if (_material_properties.get_state_ratio(
311  cell, MaterialStates::State::solid) > 0.99)
312  {
313  // Only enable the cell if it is also enabled for the thermal simulation
314  // Get the thermal DoFHandler cell iterator
315  dealii::DoFCellAccessor<dim, dim, false> thermal_cell(
316  &(_dof_handler.get_triangulation()), cell->level(), cell->index(),
317  &thermal_dof_handler);
318  auto updated_fe_index = thermal_cell.active_fe_index();
319  if (current_fe_index == updated_fe_index)
320  {
321  // The cells is unchanged, we just copy the plastic variables as-is.
322  tmp_plastic_internal_variable.push_back(
323  _plastic_internal_variable[cell_id]);
324  tmp_stress.push_back(_stress[cell_id]);
325  tmp_back_stress.push_back(_back_stress[cell_id]);
326  }
327  else
328  {
329  // The cell has solidified or material has been added. The new cells
330  // are initialized with default values.
331  auto elastic_limit = _material_properties.get_mechanical_property(
332  cell, StateProperty::elastic_limit);
333  tmp_plastic_internal_variable.push_back(
334  std::vector<double>(n_quad_pts, elastic_limit));
335  tmp_stress.push_back(
336  std::vector<dealii::SymmetricTensor<2, dim>>(n_quad_pts));
337  tmp_back_stress.push_back(
338  std::vector<dealii::SymmetricTensor<2, dim>>(n_quad_pts));
339 
340  cell->set_active_fe_index(updated_fe_index);
341  }
342  }
343  else
344  {
345  // The cell is liquid. We don't need to save the plastic variables.
346  cell->set_active_fe_index(1);
347  tmp_plastic_internal_variable.push_back(std::vector<double>(
348  n_quad_pts, std::numeric_limits<double>::signaling_NaN()));
349  tmp_stress.push_back(
350  std::vector<dealii::SymmetricTensor<2, dim>>(n_quad_pts));
351  tmp_back_stress.push_back(
352  std::vector<dealii::SymmetricTensor<2, dim>>(n_quad_pts));
353  }
354  }
355  else
356  {
357  tmp_plastic_internal_variable.push_back(std::vector<double>(
358  n_quad_pts, std::numeric_limits<double>::signaling_NaN()));
359  tmp_stress.push_back(
360  std::vector<dealii::SymmetricTensor<2, dim>>(n_quad_pts));
361  tmp_back_stress.push_back(
362  std::vector<dealii::SymmetricTensor<2, dim>>(n_quad_pts));
363  }
364  ++cell_id;
365  }
366  _plastic_internal_variable.swap(tmp_plastic_internal_variable);
367  _stress.swap(tmp_stress);
368  _back_stress.swap(tmp_back_stress);
369 
370  setup_dofs(body_forces);
371 
372  // Update _old_displacement if necessary
373  const dealii::IndexSet locally_relevant_dofs =
374  dealii::DoFTools::extract_locally_relevant_dofs(_dof_handler);
375  const dealii::IndexSet locally_owned_dofs = _dof_handler.locally_owned_dofs();
376  _old_displacement.reinit(locally_owned_dofs, locally_relevant_dofs,
377  _dof_handler.get_communicator());
378 
379  if (saved_old_displacement.size())
380  {
381  cell_id = 0;
382  for (auto const &cell : _dof_handler.active_cell_iterators())
383  {
384  if (cell->is_locally_owned())
385  {
386  auto fe_index = cell->active_fe_index();
387  if (fe_index == 0)
388  {
389  cell->get_dof_indices(global_dof_indices);
390  for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
391  {
392  if (locally_owned_dofs.is_element(global_dof_indices[i]))
393  _old_displacement[global_dof_indices[i]] =
394  saved_old_displacement[cell_id][i];
395  }
396  }
397  }
398  ++cell_id;
399  }
400  _old_displacement.compress(dealii::VectorOperation::insert);
401  }
402 }

404 template <int dim, int p_order, typename MaterialStates,
405  typename MemorySpaceType>
406 dealii::LA::distributed::Vector<double, dealii::MemorySpace::Host>
407 MechanicalPhysics<dim, p_order, MaterialStates, MemorySpaceType>::solve()
408 {
409  dealii::IndexSet locally_relevant_dofs =
410  dealii::DoFTools::extract_locally_relevant_dofs(_dof_handler);
411  dealii::LA::distributed::Vector<double, dealii::MemorySpace::Host>
412  displacement(_dof_handler.locally_owned_dofs(), locally_relevant_dofs,
413  _mechanical_operator->rhs().get_mpi_communicator());
414 
415  // Solve the mechanical problem assuming that the deformation is elastic
416  // TODO check that we are computing only difference of the displacement
417  // compared to the previous time step!!
418  unsigned int const max_iter = _dof_handler.n_dofs() / 10;
419  double const tol = 1e-12 * _mechanical_operator->rhs().l2_norm();
420  dealii::SolverControl solver_control(max_iter, tol);
421  dealii::SolverCG<
422  dealii::LA::distributed::Vector<double, dealii::MemorySpace::Host>>
423  cg(solver_control);
424  // FIXME Use better preconditioner
425  dealii::TrilinosWrappers::PreconditionSSOR preconditioner;
426  preconditioner.initialize(_mechanical_operator->system_matrix());
427  cg.solve(_mechanical_operator->system_matrix(), displacement,
428  _mechanical_operator->rhs(), preconditioner);
429  _affine_constraints.distribute(displacement);
430 
431  // Compute the new stress assuming the deformation is elastic.
432  // If the stress is under the yield criterion, the deformation is elastic and
433  // we are done. Otherwise we need to use the radial return algorithm to
434  // compute the plastic deformation.
435  dealii::LA::distributed::Vector<double, dealii::MemorySpace::Host>
436  incremental_displacement(
437  _dof_handler.locally_owned_dofs(), locally_relevant_dofs,
438  _mechanical_operator->rhs().get_mpi_communicator());
439  incremental_displacement = displacement;
440  if (_old_displacement.size() > 0)
441  {
442  incremental_displacement -= _old_displacement;
443  }
444  incremental_displacement.update_ghost_values();
445  compute_stress(incremental_displacement);
446 
447  _old_displacement.swap(displacement);
448 
449  return _old_displacement;
450 }

452 template <int dim, int p_order, typename MaterialStates,
453  typename MemorySpaceType>
454 void MechanicalPhysics<dim, p_order, MaterialStates, MemorySpaceType>::
455  compute_stress(
456  dealii::LA::distributed::Vector<double, dealii::MemorySpace::Host> const
457  &displacement)
458 {
459  dealii::hp::FEValues<dim> displacement_hp_fe_values(
460  _fe_collection, _q_collection, dealii::update_gradients);
461  unsigned int const dofs_per_cell = _fe_collection.max_dofs_per_cell();
462  std::vector<dealii::types::global_dof_index> local_dof_indices(dofs_per_cell);
463  unsigned int const n_q_points = _q_collection.max_n_quadrature_points();
464  std::vector<dealii::SymmetricTensor<2, dim>> strain_tensor(n_q_points);
465  const dealii::FEValuesExtractors::Vector displacement_extr(0);
466  unsigned int cell_id = 0;
467  for (auto const &cell : _dof_handler.active_cell_iterators())
468  {
469  if (cell->is_locally_owned() && cell->active_fe_index() == 0)
470  {
471  // Formulation based on the combined isotropic-kinematic hardening model
472  // for J2 plasticity in Chapter 3 of R. Borja, Plasticity: Modeling and
473  // Computation, Springer-Verlag, 2013. DOI: 10.1007/978-3-642-38547-6
474  //
475  // Compute the strain. We get the strain for all the quadrature points at
476  // once.
477  displacement_hp_fe_values.reinit(cell);
478  auto const &fe_values = displacement_hp_fe_values.get_present_fe_values();
479  cell->get_dof_indices(local_dof_indices);
480 
481  fe_values[displacement_extr].get_function_symmetric_gradients(
482  displacement, strain_tensor);
483 
484  double const lambda = _material_properties.get_mechanical_property(
485  cell, StateProperty::lame_first_parameter);
486  double const mu = _material_properties.get_mechanical_property(
487  cell, StateProperty::lame_second_parameter);
488  double const plastic_modulus =
489  _material_properties.get_mechanical_property(
490  cell, StateProperty::plastic_modulus);
491  double const iso_hardening_coef =
492  _material_properties.get_mechanical_property(
493  cell, StateProperty::isotropic_hardening);
494  dealii::SymmetricTensor<4, dim> stiffness_tensor =
495  lambda * dealii::outer_product(dealii::unit_symmetric_tensor<dim>(),
496  dealii::unit_symmetric_tensor<dim>()) +
497  2 * mu * dealii::identity_tensor<dim>();
498  // Loop over the quadrature points.
499  for (auto const q : fe_values.quadrature_point_indices())
500  {
501  // Compute the trial elastic stress.
502  dealii::SymmetricTensor<2, dim> elastic_stress = _stress[cell_id][q];
503  elastic_stress += stiffness_tensor * strain_tensor[q];
504 
505  auto stress_deviator = dealii::deviator(elastic_stress);
506  auto effective_stress = stress_deviator - _back_stress[cell_id][q];
507  double const effective_stress_norm = effective_stress.norm();
508  if (effective_stress_norm < _plastic_internal_variable[cell_id][q])
509  {
510  // The deformation is elastic. We just update the stress with the
511  // elastic stress.
512  _stress[cell_id][q] = elastic_stress;
513  }
514  else
515  {
516  // The deformation is plastic. We need to compute a new stress and
517  // update the plastic internal variable and the back stress.
518  double plastic_strain_increment =
519  (effective_stress_norm - _plastic_internal_variable[cell_id][q]) /
520  (2. * mu + plastic_modulus);
521  auto plastic_flow_direction =
522  effective_stress / effective_stress_norm;
523  // Update stress
524  _stress[cell_id][q] = elastic_stress - 2. * mu *
525  plastic_strain_increment *
526  plastic_flow_direction;
527  // Update plastic internal variable
528  _plastic_internal_variable[cell_id][q] +=
529  iso_hardening_coef * plastic_modulus * plastic_strain_increment;
530  // Update back stress
531  _back_stress[cell_id][q] += (1. - iso_hardening_coef) *
532  plastic_modulus * plastic_modulus *
533  plastic_flow_direction;
534  }
535  }
536  }
537  ++cell_id;
538  }
539 }
