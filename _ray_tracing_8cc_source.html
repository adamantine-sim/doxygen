95 unsigned int RayTracing::read_next_frame()
96 {
97  _rays_current_frame.clear();
98  _values_current_frame.clear();
99  for (unsigned int camera_id = _first_camera_id;
100  camera_id < _last_camera_id + 1; ++camera_id)
101  {
102  // Use regex to get the next file to read
103  std::regex camera_regex("#camera");
104  std::regex frame_regex("#frame");
105  auto filename =
106  std::regex_replace((std::regex_replace(_data_filename, camera_regex,
107  std::to_string(camera_id))),
108  frame_regex, std::to_string(_next_frame));
109  wait_for_file(filename, "Waiting for the next frame: " + filename);
110 
111  // Read and parse the file
112  std::ifstream file;
113  file.open(filename);
114  std::string line;
115  std::getline(file, line); // skip the header
116  while (std::getline(file, line))
117  {
118  std::size_t pos = 0;
119  std::size_t last_pos = 0;
120  std::size_t line_length = line.length();
121  unsigned int i = 0;
122  dealii::Point<dim> point;
123  dealii::Tensor<1, dim> direction;
124  double value = 0.;
125  while (last_pos < line_length + 1)
126  {
127  pos = line.find_first_of(',', last_pos);
128  // If no comma was found that we read until the end of the file
129  if (pos == std::string::npos)
130  {
131  pos = line_length;
132  }
133 
134  if (pos != last_pos)
135  {
136  char *end = line.data() + pos;
137  if (i < dim)
138  {
139  point[i] = std::strtod(line.data() + last_pos, &end);
140  }
141  else if (i < 2 * dim)
142  {
143  // Calculate the direction from the first and second points in the
144  // file
145  direction[i - dim] =
146  std::strtod(line.data() + last_pos, &end) - point[i - dim];
147  }
148  else
149  {
150  value = std::strtod(line.data() + last_pos, &end);
151  }
152 
153  ++i;
154  }
155 
156  last_pos = pos + 1;
157  }
158 
159  Ray<dim> ray{point, direction};
160  _rays_current_frame.push_back(ray);
161  _values_current_frame.push_back(value);
162  }
163  }
164 
165  return _next_frame++;
166 }

168 PointsValues<3> RayTracing::get_points_values()
169 {
170  // Perform the ray tracing to get the cells that are intersected by rays
171 
172  // Create the bounding boxes associated to the locally owned cells with FE
173  // index = 0
174  std::vector<dealii::BoundingBox<dim>> bounding_boxes;
175  std::vector<typename dealii::DoFHandler<dim>::active_cell_iterator>
176  cell_iterators;
177  for (auto const &cell : dealii::filter_iterators(
178  _dof_handler.active_cell_iterators(),
179  dealii::IteratorFilters::LocallyOwnedCell(),
180  dealii::IteratorFilters::ActiveFEIndexEqualTo(0)))
181  {
182  bounding_boxes.push_back(cell->bounding_box());
183  cell_iterators.push_back(cell);
184  }
185 
186  // Use ArborX to find where the rays intersect the activated cells. All the
187  // processors have access to all the rays but we still need to use
188  // DistributedTree because some rays can be stopped by activated cells on a
189  // different processors. Since the rays are on all the processors, we don't
190  // need to communicate the results to other processors.
191  auto communicator = _dof_handler.get_communicator();
192  dealii::ArborXWrappers::DistributedTree distributed_tree(communicator,
193  bounding_boxes);
194  RayNearestPredicate ray_nearest(_rays_current_frame);
195  auto [indices_ranks, offset] = distributed_tree.query(ray_nearest);
196 
197  // Find the exact intersections points
198  // See https://en.wikipedia.org/wiki/Line%E2%80%93plane_intersection
199  // NOTE that we assume that the faces are flat. If the faces are curved,
200  // this is wrong.
201  int const my_rank = dealii::Utilities::MPI::this_mpi_process(communicator);
202  unsigned int const n_rays = _rays_current_frame.size();
203  unsigned int n_intersections = 0;
204  for (unsigned int i = 0; i < n_rays; ++i)
205  {
206  for (int j = offset[i]; j < offset[i + 1]; ++j)
207  {
208  if (indices_ranks[j].second == my_rank)
209  {
210  ++n_intersections;
211  }
212  }
213  }
214 
215  std::vector<dealii::Point<dim>> points;
216  std::vector<double> values;
217  if (n_intersections != 0)
218  {
219  points.reserve(n_intersections);
220  values.reserve(n_intersections);
221  auto constexpr reference_cell =
222  dealii::ReferenceCells::get_hypercube<dim>();
223  double constexpr tol = 1e-10;
224  for (unsigned int i = 0; i < n_rays; ++i)
225  {
226  for (int j = offset[i]; j < offset[i + 1]; ++j)
227  {
228  if (indices_ranks[j].second == my_rank)
229  {
230  double distance = std::numeric_limits<double>::max();
231  dealii::Point<dim> intersection;
232  auto const &cell = cell_iterators[indices_ranks[j].first];
233  // We know that the ray intersects the bounding box but we don't know
234  // where it intersects the cells. We need to check the intersection of
235  // the ray with each face of the cell.
236  for (unsigned int f = 0; f < reference_cell.n_faces(); ++f)
237  {
238  // First we check if the ray is parallel to the face. If this is the
239  // case, either the ray misses the face or the ray hits the edge of
240  // the face. In that last case, the ray is also orthogonal to
241  // another face and it is safe to discard all rays parallel to a
242  // face.
243  auto const point_0 = cell->face(f)->vertex(0);
244  auto const point_1 = cell->face(f)->vertex(1);
245  auto const point_2 = cell->face(f)->vertex(2);
246  dealii::Tensor<1, dim> edge_01({point_1[0] - point_0[0],
247  point_1[1] - point_0[1],
248  point_1[2] - point_0[2]});
249  dealii::Tensor<1, dim> edge_02({point_2[0] - point_0[0],
250  point_2[1] - point_0[1],
251  point_2[2] - point_0[2]});
252  auto const &ray_direction = _rays_current_frame[i].direction;
253  dealii::Tensor<2, dim> matrix(
254  {{-ray_direction[0], -ray_direction[1], -ray_direction[2]},
255  {edge_01[0], edge_01[1], edge_01[2]},
256  {edge_02[0], edge_02[1], edge_02[2]}});
257  double const det = dealii::determinant(matrix);
258  double face_area = 0.;
259  for (int e = 0; e < dim; ++e)
260  for (int k = 0; k < dim; ++k)
261  face_area += std::abs(edge_01[e] * edge_02[k]);
262  // If determinant is close to zero, the ray is parallel to the face
263  // and we go to the next face.
264  if (std::abs(det) < tol * face_area)
265  continue;
266  // Compute the distance along the ray direction between the origin
267  // of the ray and the intersection point.
268  auto const cross_product =
269  dealii::cross_product_3d(edge_01, edge_02);
270  auto const &ray_origin = _rays_current_frame[i].origin;
271  dealii::Tensor<1, dim> p0_ray({ray_origin[0] - point_0[0],
272  ray_origin[1] - point_0[1],
273  ray_origin[2] - point_0[2]});
274  double d = cross_product * p0_ray / det;
275  // If d is less than 0, this means that the ray intersects the plane
276  // of the face but not the face itself. We can exit early.
277  if (d < 0)
278  continue;
279  // We can finally compute the intersection point. It is possible
280  // that a ray intersects multiple faces. For instance if the mesh is
281  // a cube the ray will get into the cube from one face and it will
282  // get out of the cube by the opposite face. The correct
283  // intersection point is the one with the smallest distance.
284  if (d < distance)
285  {
286  // The point intersects the plane of the face but maybe not the
287  // face itself. Check that the point is on the face. NOTE: We
288  // assume that the face is an axis-aligned rectangle.
289  dealii::Point<dim> face_intersection =
290  ray_origin + d * ray_direction;
291  std::vector<double> min(dim, std::numeric_limits<double>::max());
292  std::vector<double> max(dim,
293  std::numeric_limits<double>::lowest());
294  for (unsigned int coord = 0; coord < dim; ++coord)
295  {
296  if (point_0[coord] < min[coord])
297  min[coord] = point_0[coord];
298  if (point_0[coord] > max[coord])
299  max[coord] = point_0[coord];
300 
301  if (point_1[coord] < min[coord])
302  min[coord] = point_1[coord];
303  if (point_1[coord] > max[coord])
304  max[coord] = point_1[coord];
305 
306  if (point_2[coord] < min[coord])
307  min[coord] = point_2[coord];
308  if (point_2[coord] > max[coord])
309  max[coord] = point_2[coord];
310  }
311 
312  bool on_the_face = true;
313  double const effective_edge = std::sqrt(face_area);
314  for (int coord = 0; coord < dim; ++coord)
315  {
316  if ((face_intersection[coord] <
317  (min[coord] - tol * effective_edge)) ||
318  (face_intersection[coord] >
319  (max[coord] + tol * effective_edge)))
320  {
321  on_the_face = false;
322  break;
323  }
324  }
325 
326  if (on_the_face)
327  {
328  distance = d;
329  intersection = face_intersection;
330  }
331  }
332  }
333  if (distance < std::numeric_limits<double>::max())
334  {
335  points.push_back(intersection);
336  values.push_back(_values_current_frame[i]);
337  }
338  }
339  }
340  }
341  // We know that there are at most n_intersections between the rays and the
342  // mesh
343  ASSERT(points.size() <= n_intersections,
344  "Error when computing the intersection of rays with the mesh.");
345  }
346 
347  // We need all the processors to know the intersection points of the rays with
348  // the mesh and the values at these points.
349  PointsValues<dim> points_values;
350  auto all_points = dealii::Utilities::MPI::all_gather(communicator, points);
351  auto all_values = dealii::Utilities::MPI::all_gather(communicator, values);
352  for (unsigned int i = 0; i < all_points.size(); ++i)
353  {
354  points_values.points.insert(points_values.points.end(),
355  all_points[i].begin(), all_points[i].end());
356  points_values.values.insert(points_values.values.end(),
357  all_values[i].begin(), all_values[i].end());
358  }
359 
360  return points_values;
361 }
