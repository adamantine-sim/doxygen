158 template <int dim>
159 void Geometry<dim>::assign_material_state(
160  boost::property_tree::ptree const &database)
161 {
162  // PropertyTreeInput geometry.material_height
163  double const material_height = database.get("material_height", 1e9);
164  // PropertyTreeInput geometry.use_powder
165  bool const use_powder = database.get("use_powder", false);
166 
167  if (use_powder)
168  {
169  // PropertyTreeInput geometry.powder_layer
170  double const powder_layer = database.get<double>("powder_layer");
171  double const solid_height = material_height - powder_layer;
172 
173  for (auto cell : _triangulation.active_cell_iterators())
174  {
175  if (cell->center()[axis<dim>::z] < solid_height)
176  {
177  cell->set_user_index(static_cast<int>(SolidLiquidPowder::State::solid));
178  }
179  else
180  {
181  cell->set_user_index(
182  static_cast<int>(SolidLiquidPowder::State::powder));
183  }
184  }
185  }
186  else
187  {
188  // Everything is made of solid material
189  for (auto cell : _triangulation.active_cell_iterators())
190  {
191  cell->set_user_index(static_cast<int>(SolidLiquidPowder::State::solid));
192  }
193  }
194 }
