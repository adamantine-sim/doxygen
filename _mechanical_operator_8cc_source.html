109 template <int dim, int p_order, typename MaterialStates,
110  typename MemorySpaceType>
111 void MechanicalOperator<dim, p_order, MaterialStates, MemorySpaceType>::
112  assemble_system(
113  std::vector<std::shared_ptr<BodyForce<dim>>> const &body_forces)
114 {
115  // Create the sparsity pattern. Since we use a Trilinos matrix we don't need
116  // the sparsity pattern to outlive the sparse matrix.
117  auto locally_owned_dofs = _dof_handler->locally_owned_dofs();
118  auto locally_relevant_dofs =
119  dealii::DoFTools::extract_locally_relevant_dofs(*_dof_handler);
120  dealii::DynamicSparsityPattern dsp(locally_relevant_dofs);
121  dealii::DoFTools::make_sparsity_pattern(*_dof_handler, dsp,
122  *_affine_constraints, false);
123  dealii::SparsityTools::distribute_sparsity_pattern(
124  dsp, locally_owned_dofs, _communicator, locally_relevant_dofs);
125 
126  _system_matrix.reinit(locally_owned_dofs, dsp, _communicator);
127 
128  dealii::hp::FEValues<dim> displacement_hp_fe_values(
129  _dof_handler->get_fe_collection(), *_q_collection,
130  dealii::update_values | dealii::update_gradients |
131  dealii::update_JxW_values);
132 
133  unsigned int const dofs_per_cell =
134  _dof_handler->get_fe_collection().max_dofs_per_cell();
135  dealii::FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
136  std::vector<dealii::types::global_dof_index> local_dof_indices(dofs_per_cell);
137 
138  // Loop over the locally owned cells that are not FE_Nothing and assemble the
139  // sparse matrix and the right-hand-side
140  for (auto const &cell : _dof_handler->active_cell_iterators() |
141  dealii::IteratorFilters::ActiveFEIndexEqualTo(
142  0, /* locally owned */ true))
143  {
144  displacement_hp_fe_values.reinit(cell);
145  auto const &fe_values = displacement_hp_fe_values.get_present_fe_values();
146  auto const &fe = fe_values.get_fe();
147 
148  // Assemble the local martrix
149  cell_matrix = 0;
150  double const lambda = this->_material_properties.get_mechanical_property(
151  cell, StateProperty::lame_first_parameter);
152  double const mu = this->_material_properties.get_mechanical_property(
153  cell, StateProperty::lame_second_parameter);
154  for (auto const i : fe_values.dof_indices())
155  {
156  auto const component_i = fe.system_to_component_index(i).first;
157  for (auto const j : fe_values.dof_indices())
158  {
159  auto const component_j = fe.system_to_component_index(j).first;
160  for (auto const q_point : fe_values.quadrature_point_indices())
161  {
162  cell_matrix(i, j) +=
163  // FIXME We should be able to use the following formulation but
164  // the result is different. We need to understand why.
165  // ((lambda + mu) * fe_values.shape_grad(i, q_point)[component_i]
166  // * fe_values.shape_grad(j, q_point)[component_j] +
167  ((fe_values.shape_grad(i, q_point)[component_i] *
168  fe_values.shape_grad(j, q_point)[component_j] * lambda) +
169  (fe_values.shape_grad(i, q_point)[component_j] *
170  fe_values.shape_grad(j, q_point)[component_i] * mu) +
171  ((component_i == component_j)
172  ? mu * fe_values.shape_grad(i, q_point) *
173  fe_values.shape_grad(j, q_point)
174  : 0.)) *
175  fe_values.JxW(q_point);
176  }
177  }
178  }
179  cell->get_dof_indices(local_dof_indices);
180  _affine_constraints->distribute_local_to_global(
181  cell_matrix, local_dof_indices, _system_matrix);
182  }
183 
184  // Assemble the rhs
185  dealii::LA::distributed::Vector<double, dealii::MemorySpace::Host>
186  assembled_rhs(locally_owned_dofs, locally_relevant_dofs, _communicator);
187  dealii::Vector<double> cell_rhs(dofs_per_cell);
188  // If the list of reference temperatures is non-empty, we solve the
189  // thermo-elastic problem.
190  if (_reference_temperatures.size() > 0)
191  {
192  // Create temperature hp::FEValues using the same finite elements as the
193  // thermal simulation but evaluated at the quadrature points of the
194  // mechanical simulations.
195  dealii::hp::FEValues<dim> temperature_hp_fe_values(
196  _thermal_dof_handler->get_fe_collection(), *_q_collection,
197  dealii::update_values);
198 
199  // _has_melted is using its own indices. The indices are computed using the
200  // locally owned cells of the thermal DoFHandler that have an FE_Index
201  // equal to zero. We need to translate these indices to be used with the
202  // mechanical DoFHandler. This is simplified by the fact that both the
203  // thermal and the mechanical simulation use the same Triangulation and have
204  // the same cells locally owned.
205  auto &triangulation = _dof_handler->get_triangulation();
206  unsigned int const n_local_active_cells = triangulation.n_active_cells();
207  std::vector<unsigned int> cell_indices(n_local_active_cells);
208  unsigned int cell_index = 0;
209  for (auto const &tria_cell :
210  triangulation.active_cell_iterators() |
211  dealii::IteratorFilters::LocallyOwnedCell())
212  {
213  dealii::TriaIterator<dealii::DoFCellAccessor<dim, dim, false>>
214  temperature_cell(&triangulation, tria_cell->level(),
215  tria_cell->index(), _thermal_dof_handler);
216  if (temperature_cell->active_fe_index() == 0)
217  {
218  dealii::TriaIterator<dealii::DoFCellAccessor<dim, dim, false>>
219  displacement_cell(&triangulation, tria_cell->level(),
220  tria_cell->index(), _dof_handler);
221  if (displacement_cell->active_fe_index() == 0)
222  {
223  cell_indices[displacement_cell->active_cell_index()] = cell_index;
224  }
225  ++cell_index;
226  }
227  }
228 
229  _temperature.update_ghost_values();
230 
231  std::vector<dealii::types::global_dof_index> temperature_local_dof_indices(
232  _thermal_dof_handler->get_fe_collection().max_dofs_per_cell());
233  double const initial_temperature = _reference_temperatures.back();
234  for (auto const &cell : _dof_handler->active_cell_iterators() |
235  dealii::IteratorFilters::ActiveFEIndexEqualTo(
236  0, /* locally owned */ true))
237  {
238  cell_rhs = 0.;
239 
240  // Get the temperature cell associated to the mechanical cell
241  dealii::TriaIterator<dealii::DoFCellAccessor<dim, dim, false>>
242  temperature_cell(&triangulation, cell->level(), cell->index(),
243  _thermal_dof_handler);
244 
245  // Get the appropriate reference temperature for the cell. If the cell
246  // is not in the unmelted substrate, the reference temperature depends
247  // on the material.
248  double reference_temperature =
249  _has_melted[cell_indices[cell->active_cell_index()]]
250  ? _reference_temperatures[temperature_cell->material_id()]
251  : initial_temperature;
252 
253  displacement_hp_fe_values.reinit(cell);
254  auto const &fe_values = displacement_hp_fe_values.get_present_fe_values();
255 
256  double const alpha = this->_material_properties.get_mechanical_property(
257  cell, StateProperty::thermal_expansion_coef);
258  double const lambda = this->_material_properties.get_mechanical_property(
259  cell, StateProperty::lame_first_parameter);
260  double const mu = this->_material_properties.get_mechanical_property(
261  cell, StateProperty::lame_second_parameter);
262 
263  temperature_hp_fe_values.reinit(temperature_cell);
264  auto &temperature_fe_values =
265  temperature_hp_fe_values.get_present_fe_values();
266  temperature_cell->get_dof_indices(temperature_local_dof_indices);
267 
268  dealii::FEValuesExtractors::Vector const displacements(0);
269 
270  for (auto const q_point : fe_values.quadrature_point_indices())
271  {
272  double delta_T = -reference_temperature;
273  for (unsigned int j = 0; j < temperature_fe_values.dofs_per_cell; ++j)
274  {
275  delta_T += temperature_fe_values.shape_value(j, q_point) *
276  _temperature(temperature_local_dof_indices[j]);
277  }
278  auto B = dealii::Physics::Elasticity::StandardTensors<dim>::I;
279  B *= (3. * lambda + 2 * mu) * alpha * delta_T;
280 
281  for (auto const i : fe_values.dof_indices())
282  {
283  cell_rhs(i) += dealii::scalar_product(
284  B, fe_values[displacements].gradient(i, q_point)) *
285  fe_values.JxW(q_point);
286  }
287  }
288 
289  cell->get_dof_indices(local_dof_indices);
290  _affine_constraints->distribute_local_to_global(
291  cell_rhs, local_dof_indices, assembled_rhs);
292  }
293  }
294 
295  // Add gravitational body force
296  if (body_forces.size())
297  {
298  for (auto const &cell : _dof_handler->active_cell_iterators() |
299  dealii::IteratorFilters::ActiveFEIndexEqualTo(
300  0, /* locally owned */ true))
301  {
302  cell_rhs = 0.;
303 
304  displacement_hp_fe_values.reinit(cell);
305  auto const &fe_values = displacement_hp_fe_values.get_present_fe_values();
306 
307  dealii::Tensor<1, dim, double> body_force;
308  for (auto &force : body_forces)
309  {
310  body_force += force->eval(cell);
311  }
312 
313  dealii::FEValuesExtractors::Vector const displacements(0);
314 
315  for (auto const i : fe_values.dof_indices())
316  {
317  for (auto const q_point : fe_values.quadrature_point_indices())
318  cell_rhs(i) += body_force *
319  fe_values[displacements].value(i, q_point) *
320  fe_values.JxW(q_point);
321  }
322 
323  cell->get_dof_indices(local_dof_indices);
324  _affine_constraints->distribute_local_to_global(
325  cell_rhs, local_dof_indices, assembled_rhs);
326  }
327  }
328 
329  _system_matrix.compress(dealii::VectorOperation::add);
330  assembled_rhs.compress(dealii::VectorOperation::add);
331 
332  // When solving the system, we don't want ghost entries
333  _system_rhs.reinit(_dof_handler->locally_owned_dofs(), _communicator);
334  _system_rhs = assembled_rhs;
335 }
