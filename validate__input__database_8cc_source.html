16 {
17  // Tree: physics
18  // PropertyTreeInput physics.thermal
19  bool const use_thermal_physics = database.get<bool>("physics.thermal");
20  // PropertyTreeInput physics.mechanical
21  bool const use_mechanical_physics = database.get<bool>("physics.mechanical");
22  ASSERT_THROW(use_thermal_physics || use_mechanical_physics,
23  "Error: both thermal and mechanical physics are disabled");
24 
25  // Tree: boundary
26  // Read the boundary condition type and check for disallowed combinations
27  // PropertyTreeInput boundary.type
28  BoundaryType boundary_type = BoundaryType::invalid;
29  std::string boundary_type_str = database.get<std::string>("boundary.type");
30  // Parse the string
31  size_t pos_str = 0;
32  std::string boundary;
33  std::string delimiter = ",";
34  auto parse_boundary_type = [&](std::string const &boundary)
35  {
36  if (boundary == "adiabatic")
37  {
38  ASSERT_THROW(
39  boundary_type == BoundaryType::invalid,
40  "Error: Adiabatic condition cannot be combined with another type.");
41  boundary_type = BoundaryType::adiabatic;
42  }
43  else
44  {
45  ASSERT_THROW(
46  boundary_type != BoundaryType::adiabatic,
47  "Error: Adiabatic condition cannot be combined with another type.");
48 
49  if (boundary == "radiative")
50  {
51  boundary_type |= BoundaryType::radiative;
52  }
53  else if (boundary == "convective")
54  {
55  boundary_type |= BoundaryType::convective;
56  }
57  else
58  {
59  ASSERT_THROW(false, "Error: Unknown boundary type.");
60  }
61  }
62  };
63  while ((pos_str = boundary_type_str.find(delimiter)) != std::string::npos)
64  {
65  boundary = boundary_type_str.substr(0, pos_str);
66  parse_boundary_type(boundary);
67  boundary_type_str.erase(0, pos_str + delimiter.length());
68  }
69  parse_boundary_type(boundary_type_str);

72  if (use_thermal_physics)
73  {
74  // PropertyTreeInput discretization.thermal.fe_degree
75  unsigned int const fe_degree =
76  database.get<unsigned int>("discretization.thermal.fe_degree");
77  ASSERT_THROW(fe_degree > 0 && fe_degree < 6,
78  "Error: fe_degree should be between 1 and 5.");
79

82  database.get_optional<std::string>("discretization.thermal.quadrature");
83 
84  if (quadrature_type_optional)

87  if (!((boost::iequals(quadrature_type, "gauss") ||
88  (boost::iequals(quadrature_type, "lobatto")))))

90  ASSERT_THROW(false, "Error: Unknown quadrature type.");
91  }
92  }

105  "Error: Powder layer must be non-negative.");

107

109  database.get("geometry.material_deposition", false);

117  "Error: Method type for material deposition, '" + method +
118  "', is not recognized. Valid options are: 'file' "
119  "and 'scan_paths'");

124  "Error: If the material deposition method is 'file', "

126  }

135  "Error: If the material deposition method is 'scan_path', "
136  "'deposition_height' must be given.");

147  "'deposition_lead_time' must be given.");
148  }
149  }
