125 template <int dim, bool use_table, int p_order, int fe_degree,
126  typename MaterialStates>
127 KOKKOS_FUNCTION void
128 ThermalOperatorQuad<dim, use_table, p_order, fe_degree, MaterialStates>::
129  update_state_ratios([[maybe_unused]] unsigned int pos,
130  [[maybe_unused]] double temperature,
131  double *state_ratios) const
132 {
133  unsigned int constexpr solid =
134  static_cast<unsigned int>(MaterialStates::State::solid);
135 
136  if constexpr (std::is_same_v<MaterialStates, adamantine::Solid>)
137  {
138  state_ratios[solid] = 1.;
139  }
140  else
141  {
142  // Get the material id at this point
143  auto material_id = _material_id[pos];
144 
145  // Get the material thermodynamic properties
146  double const solidus = _properties(
147  material_id, static_cast<unsigned int>(adamantine::Property::solidus));
148  double const liquidus = _properties(
149  material_id, static_cast<unsigned int>(adamantine::Property::liquidus));
150 
151  unsigned int constexpr liquid =
152  static_cast<unsigned int>(MaterialStates::State::liquid);
153 
154  if constexpr (std::is_same_v<MaterialStates, adamantine::SolidLiquid>)
155  {
156  // Update the state ratios
157  if (temperature < solidus)
158  {
159  state_ratios[liquid] = 0.;
160  }
161  else if (temperature > liquidus)
162  {
163  state_ratios[liquid] = 1.;
164  }
165  else
166  {
167  state_ratios[liquid] = (temperature - solidus) / (liquidus - solidus);
168  }
169 
170  // round-off.
171  state_ratios[solid] = 1. - state_ratios[liquid];
172 
173  _liquid_ratio(pos) = state_ratios[liquid];
174  }
175  else if constexpr (std::is_same_v<MaterialStates,
176  adamantine::SolidLiquidPowder>)
177  {
178  unsigned int constexpr powder =
179  static_cast<unsigned int>(MaterialStates::State::powder);
180 
181  // Update the state ratios
182  state_ratios[powder] = _powder_ratio(pos);
183 
184  if (temperature < solidus)
185  state_ratios[liquid] = 0.;
186  else if (temperature > liquidus)
187  state_ratios[liquid] = 1.;
188  else
189  state_ratios[liquid] = (temperature - solidus) / (liquidus - solidus);
190 
191  // Because the powder can only become liquid, the solid can only
192  // become liquid, and the liquid can only become solid, the ratio of
193  // powder can only decrease.
194  state_ratios[powder] = (1. - state_ratios[liquid]) < state_ratios[powder]
195  ? (1. - state_ratios[liquid])
196  : state_ratios[powder];
197 
198  state_ratios[solid] = 1. - state_ratios[liquid] - state_ratios[powder];
199 
200  _powder_ratio(pos) = state_ratios[powder];
201  _liquid_ratio(pos) = state_ratios[liquid];
202  }
203  }
204 }

206 template <int dim, bool use_table, int p_order, int fe_degree,
207  typename MaterialStates>
208 KOKKOS_FUNCTION void
209 ThermalOperatorQuad<dim, use_table, p_order, fe_degree,
210  MaterialStates>::get_inv_rho_cp(unsigned int pos,
211  double *state_ratios,
212  double temperature) const
213 {
214  // Here we need the specific heat (including the latent heat contribution)
215  // and the density
216 
217  auto material_id = _material_id(pos);
218 
219  // Compute the state-dependent properties
220  double const density =
221  compute_material_property(adamantine::StateProperty::density, material_id,
222  state_ratios, temperature);
223 
224  double specific_heat =
225  compute_material_property(adamantine::StateProperty::specific_heat,
226  material_id, state_ratios, temperature);
227 
228  if constexpr (!std::is_same_v<MaterialStates, adamantine::Solid>)
229  {
230  // Get the state-independent material properties
231  double const solidus = _properties(
232  material_id, static_cast<unsigned int>(adamantine::Property::solidus));
233  double const liquidus = _properties(
234  material_id, static_cast<unsigned int>(adamantine::Property::liquidus));
235  double const latent_heat = _properties(
236  material_id,
237  static_cast<unsigned int>(adamantine::Property::latent_heat));
238 
239  // Add in the latent heat contribution
240  unsigned int constexpr liquid =
241  static_cast<unsigned int>(MaterialStates::State::liquid);
242 
243  if (state_ratios[liquid] > 0.0 && (state_ratios[liquid] < 1.0))
244  {
245  specific_heat += latent_heat / (liquidus - solidus);
246  }
247  }
248 
249  _inv_rho_cp(pos) = 1.0 / (density * specific_heat);
250 }

252 template <int dim, bool use_table, int p_order, int fe_degree,
253  typename MaterialStates>
254 KOKKOS_FUNCTION double
255 ThermalOperatorQuad<dim, use_table, p_order, fe_degree, MaterialStates>::
256  compute_material_property(adamantine::StateProperty state_property,
257  dealii::types::material_id const material_id,
258  double const *state_ratios,
259  double temperature) const
260 {
261  double value = 0.0;
262  unsigned int const property_index = static_cast<unsigned int>(state_property);
263  if constexpr (use_table)
264  {
265  for (unsigned int material_state = 0; material_state < _n_material_states;
266  ++material_state)
267  {
268  const dealii::types::material_id m_id = material_id;
269 
270  value += state_ratios[material_state] *
271  adamantine::MaterialProperty<dim, p_order, MaterialStates,
272  dealii::MemorySpace::Default>::
273  compute_property_from_table(_state_property_tables, m_id,
274  material_state, property_index,
275  temperature);
276  }
277  }
278  else
279  {
280  for (unsigned int material_state = 0; material_state < _n_material_states;
281  ++material_state)
282  {
283  dealii::types::material_id m_id = material_id;
284 
285  for (unsigned int i = 0; i <= p_order; ++i)
286  {
287  value += state_ratios[material_state] *
288  _state_property_polynomials(m_id, material_state,
289  property_index, i) *
290  std::pow(temperature, i);
291  }
292  }
293  }
294 
295  return value;
296 }

298 template <int dim, bool use_table, int p_order, int fe_degree,
299  typename MaterialStates>
300 KOKKOS_FUNCTION void
301 ThermalOperatorQuad<dim, use_table, p_order, fe_degree, MaterialStates>::
302 operator()(dealii::Portable::FEEvaluation<dim, fe_degree> *fe_eval,
303  int const q_point) const
304 {
305  double temperature = fe_eval->get_value(q_point);
306 
307  double state_ratios[MaterialStates::n_material_states];
308  unsigned int const pos =
309  _gpu_data->local_q_point_id(_cell, _n_q_points, q_point);
310  update_state_ratios(pos, temperature, state_ratios);
311  get_inv_rho_cp(pos, state_ratios, temperature);
312  auto material_id = _material_id[pos];
313  auto const thermal_conductivity_x = compute_material_property(
314  adamantine::StateProperty::thermal_conductivity_x, material_id,
315  state_ratios, temperature);
316  auto const thermal_conductivity_y = compute_material_property(
317  adamantine::StateProperty::thermal_conductivity_y, material_id,
318  state_ratios, temperature);
319  auto const thermal_conductivity_z = compute_material_property(
320  adamantine::StateProperty::thermal_conductivity_z, material_id,
321  state_ratios, temperature);
322 
323  auto th_conductivity_grad = fe_eval->get_gradient(q_point);
324 
325  // In 2D we only use x and z, and there are no deposition angle
326  if constexpr (dim == 2)
327  {
328  th_conductivity_grad[::adamantine::axis<dim>::x] *= thermal_conductivity_x;
329  th_conductivity_grad[::adamantine::axis<dim>::z] *= thermal_conductivity_z;
330  }
331 
332  if constexpr (dim == 3)
333  {
334  auto const th_conductivity_grad_x =
335  th_conductivity_grad[::adamantine::axis<dim>::x];
336  auto const th_conductivity_grad_y =
337  th_conductivity_grad[::adamantine::axis<dim>::y];
338 
339  // The rotation is performed using the following formula
340  //
341  // (cos -sin) (x 0) ( cos sin)
342  // (sin cos) (0 y) (-sin cos)
343  // =
344  // ((x*cos^2 + y*sin^2) ((x-y) * (sin*cos)))
345  // (((x-y) * (sin*cos)) (x*sin^2 + y*cos^2))
346  th_conductivity_grad[::adamantine::axis<dim>::x] =
347  (thermal_conductivity_x * _cos[pos] * _cos[pos] +
348  thermal_conductivity_y * _sin[pos] * _sin[pos]) *
349  th_conductivity_grad_x +
350  ((thermal_conductivity_x - thermal_conductivity_y) * _sin[pos] *
351  _cos[pos]) *
352  th_conductivity_grad_y;
353  th_conductivity_grad[::adamantine::axis<dim>::y] =
354  ((thermal_conductivity_x - thermal_conductivity_y) * _sin[pos] *
355  _cos[pos]) *
356  th_conductivity_grad_x +
357  (thermal_conductivity_x * _sin[pos] * _sin[pos] +
358  thermal_conductivity_y * _cos[pos] * _cos[pos]) *
359  th_conductivity_grad_y;
360 
361  // There is no deposition angle for the z axis
362  th_conductivity_grad[::adamantine::axis<dim>::z] *= thermal_conductivity_z;
363  }
364 
365  fe_eval->submit_gradient(-_inv_rho_cp[pos] * th_conductivity_grad, q_point);
366 }

470  reinit(dealii::DoFHandler<dim> const &dof_handler,
471  dealii::AffineConstraints<double> const &affine_constraints,

475  // deal.II does not support QCollection on GPU
476  _matrix_free.reinit(dealii::StaticMappingQ1<dim>::mapping, dof_handler,
477  affine_constraints, q_collection[0], filter,

479  dealii::LA::distributed::Vector<double, MemorySpaceType> tmp;
480  _matrix_free.initialize_dof_vector(tmp);

492  dealii::Utilities::pow(n_dofs_1d, dim);
493  auto graph = _matrix_free.get_colored_graph();
494  unsigned int const n_colors = graph.size();
495  for (unsigned int color = 0; color < n_colors; ++color)

501  gpu_data, _matrix_free_data.mapping_update_flags);
502  for (unsigned int cell_id = 0; cell_id < n_cells; ++cell_id)
503  {
504  auto cell = graph[color][cell_id];
505  std::vector<unsigned int> quad_pos(n_q_points_per_cell);
506  for (unsigned int i = 0; i < n_q_points_per_cell; ++i)
507  {
508  unsigned int const pos =

511  }

513  }
514  }

517 template <int dim, bool use_table, int p_order, int fe_degree,
518  typename MaterialStates, typename MemorySpaceType>
519 void ThermalOperatorDevice<dim, use_table, p_order, fe_degree, MaterialStates,
520  MemorySpaceType>::
521  compute_inverse_mass_matrix(
522  dealii::DoFHandler<dim> const &dof_handler,
523  dealii::AffineConstraints<double> const &affine_constraints)
524 {
525  // Compute the inverse of the mass matrix
526  dealii::QGaussLobatto<1> mass_matrix_quad(fe_degree + 1);
527  dealii::Portable::MatrixFree<dim, double> mass_matrix_free;
528 
529  typename dealii::Portable::MatrixFree<dim, double>::AdditionalData mf_data;
530  // Due to a bug in deal.II 9.5 we need to update the quadrature points and the
531  // gradients
532 #if (DEAL_II_VERSION_MAJOR == 9) && (DEAL_II_VERSION_MINOR == 5)
533  mf_data.mapping_update_flags =
534  dealii::update_quadrature_points | dealii::update_values |
535  dealii::update_gradients | dealii::update_JxW_values;
536 #else
537  mf_data.mapping_update_flags =
538  dealii::update_values | dealii::update_JxW_values;
539 #endif
540  dealii::IteratorFilters::ActiveFEIndexEqualTo filter(0, true);
541  mass_matrix_free.reinit(dealii::StaticMappingQ1<dim>::mapping, dof_handler,
542  affine_constraints, mass_matrix_quad, filter,
543  mf_data);
544  mass_matrix_free.initialize_dof_vector(*_inverse_mass_matrix);
545  // We don't save memory by not allocating the vector. Instead this is done in
546  // cell_loop by using a slower path
547  dealii::LA::distributed::Vector<double, MemorySpaceType> dummy(
548  _inverse_mass_matrix->get_partitioner());
549  LocalMassMatrixOperator<dim, fe_degree> local_operator;
550  mass_matrix_free.cell_loop(local_operator, dummy, *_inverse_mass_matrix);
551  _inverse_mass_matrix->compress(dealii::VectorOperation::add);
552  unsigned int const local_size = _inverse_mass_matrix->locally_owned_size();
553  double *values = _inverse_mass_matrix->get_values();
