32 double GoldakHeatSource<dim>::value(dealii::Point<dim> const &point,
33  double const height) const
34 {
35  double const z = point[axis<dim>::z] - height;
36  if ((z + this->_beam.depth) < 0.)
37  {
38  return 0.;
39  }
40  else
41  {
42  double xpy_squared =
43  std::pow(point[axis<dim>::x] - _beam_center[axis<dim>::x], 2);
44  if (dim == 3)
45  {
46  xpy_squared +=
47  std::pow(point[axis<dim>::y] - _beam_center[axis<dim>::y], 2);
48  }
49 
50  // Goldak heat source equation
51  double heat_source =
52  _alpha * std::exp(-3.0 * xpy_squared / this->_beam.radius_squared +
53  -3.0 * std::pow(z / this->_beam.depth, 2));
54
