90 template <int dim>
91 dealii::Vector<double> PostProcessor<dim>::get_stress_norm(
92  std::vector<std::vector<dealii::SymmetricTensor<2, dim>>> const
93  &stress_tensor)
94 {
95  dealii::Vector<double> norm(
96  _mechanical_dof_handler->get_triangulation().n_active_cells());
97  unsigned int const n_quad_pts =
98  stress_tensor.size() > 0 ? stress_tensor[0].size() : 0;
99  unsigned int cell_id = 0;
100  for (auto const &cell : _mechanical_dof_handler->active_cell_iterators() |
101  dealii::IteratorFilters::ActiveFEIndexEqualTo(
102  0, /* locally owned */ true))
103  {
104  dealii::SymmetricTensor<2, dim> accumulated_stress;
105  for (unsigned int q = 0; q < n_quad_pts; ++q)
106  {
107  accumulated_stress += stress_tensor[cell_id][q];
108  }
109 
110  norm(cell->active_cell_index()) = (accumulated_stress / n_quad_pts).norm();
111 
112  ++cell_id;
113  }
114 
115  return norm;
116 }

141  displacement_names,
142  displacement_data_component_interpretation);
143 
144  // Add the strain tensor to the output
145  _data_out.add_data_vector(*_mechanical_dof_handler, displacement, strain);
146 
147  // Add the stress tensor

167 template <int dim>
168 void PostProcessor<dim>::write_pvtu(unsigned int time_step, double time)
169 {
170  dealii::DoFHandler<dim> *dof_handler =
171  (_thermal_dof_handler) ? _thermal_dof_handler : _mechanical_dof_handler;
172  dealii::types::subdomain_id subdomain_id =
173  dof_handler->get_triangulation().locally_owned_subdomain();
174  _data_out.build_patches(_additional_output_refinement);
175  std::string local_filename = _filename_prefix + "." +
176  dealii::Utilities::to_string(time_step) + "." +
177  dealii::Utilities::to_string(subdomain_id);
178  std::ofstream output((local_filename + ".vtu").c_str());
179  dealii::DataOutBase::VtkFlags flags(time);
180  _data_out.set_flags(flags);
181  _data_out.write_vtu(output);
182 
183  unsigned int rank = dealii::Utilities::MPI::this_mpi_process(_communicator);
184  if (rank == 0)
185  {
186  std::vector<std::string> filenames;
187  unsigned int comm_size =
188  dealii::Utilities::MPI::n_mpi_processes(_communicator);
189  for (unsigned int i = 0; i < comm_size; ++i)
190  {
191  std::string local_name = _filename_prefix + "." +
192  dealii::Utilities::to_string(time_step) + "." +
193  dealii::Utilities::to_string(i) + ".vtu";
194  filenames.push_back(local_name);
195  }
196  std::string pvtu_filename = _filename_prefix + "." +
197  dealii::Utilities::to_string(time_step) +
198  ".pvtu";
199  std::ofstream pvtu_output(pvtu_filename.c_str());
200  _data_out.write_pvtu_record(pvtu_output, filenames);
201 
202  // Associate the time to the time step.
203  _times_filenames.push_back(
204  std::pair<double, std::string>(time, pvtu_filename));
205  }
