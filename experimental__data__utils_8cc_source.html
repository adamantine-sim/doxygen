21 template <int dim>
22 std::pair<std::vector<dealii::types::global_dof_index>,
23  std::vector<dealii::Point<dim>>>
24 get_dof_to_support_mapping(dealii::DoFHandler<dim> const &dof_handler)
25 {
26  std::vector<dealii::types::global_dof_index> dof_indices;
27  std::vector<dealii::Point<dim>> support_points;
28  std::unordered_set<dealii::types::global_dof_index> visited_dof_indices;
29 
30  // Manually do what dealii::DoFTools::map_dofs_to_support_points does, since
31  // that doesn't currently work with FE_Nothing
32  const dealii::FiniteElement<dim> &fe = dof_handler.get_fe(0);
33 
34  dealii::FEValues<dim, dim> fe_values(fe, fe.get_unit_support_points(),
35  dealii::update_quadrature_points);
36 
37  std::vector<dealii::types::global_dof_index> local_dof_indices(
38  fe.n_dofs_per_cell());
39  auto locally_owned_dofs = dof_handler.locally_owned_dofs();
40 
41  for (auto const &cell :
42  dealii::filter_iterators(dof_handler.active_cell_iterators(),
43  dealii::IteratorFilters::ActiveFEIndexEqualTo(
44  0, /* locally owned */ true)))
45  {
46  fe_values.reinit(cell);
47  cell->get_dof_indices(local_dof_indices);
48  const std::vector<dealii::Point<dim>> &points =
49  fe_values.get_quadrature_points();
50  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
51  {
52  // Skip duplicate points like vertices and indices that correspond to
53  // ghosted elements
54  if ((visited_dof_indices.count(local_dof_indices[i]) == 0) &&
55  (locally_owned_dofs.is_element(local_dof_indices[i])))
56  {
57  dof_indices.push_back(local_dof_indices[i]);
58  support_points.push_back(points[i]);
59  visited_dof_indices.insert(local_dof_indices[i]);
60  }
61  }
62  }
63 
64  return {dof_indices, support_points};
65 }

67 template <int dim>
68 std::pair<std::vector<int>, std::vector<int>>
69 get_expt_to_dof_mapping(PointsValues<dim> const &points_values,
70  dealii::DoFHandler<dim> const &dof_handler)
71 {
72  auto [dof_indices, support_points] = get_dof_to_support_mapping(dof_handler);
73 
74  auto communicator = dof_handler.get_communicator();
75  int my_rank = dealii::Utilities::MPI::this_mpi_process(communicator);
76 
77  // Perform the search
78  dealii::ArborXWrappers::DistributedTree distributed_tree(communicator,
79  support_points);
80  dealii::ArborXWrappers::PointNearestPredicate pt_nearest(points_values.points,
81  1);
82  auto [indices_ranks, offset] = distributed_tree.query(pt_nearest);
83 
84  // Convert the indices and offsets to a pair that maps experimental indices to
85  // dof indices
86  std::vector<int> obs_indices;
87  std::vector<int> global_indices;
88  obs_indices.resize(indices_ranks.size());
89  global_indices.resize(indices_ranks.size());
90  unsigned int const n_queries = points_values.points.size();
91  for (unsigned int i = 0; i < n_queries; ++i)
92  {
93  for (int j = offset[i]; j < offset[i + 1]; ++j)
94  {
95  obs_indices[j] = i;
96  global_indices[j] = indices_ranks[j].second == my_rank
97  ? dof_indices[indices_ranks[j].first]
98  : -1;
99  }
100  }
101 
102  dealii::Utilities::MPI::max(global_indices, communicator, global_indices);
103 
104  return std::make_pair(obs_indices, global_indices);
105 }

107 template <int dim>
108 void set_with_experimental_data(
109  MPI_Comm const &communicator, PointsValues<dim> const &points_values,
110  std::pair<std::vector<int>, std::vector<int>> &expt_to_dof_mapping,
111  dealii::LinearAlgebra::distributed::Vector<double> &temperature,
112  bool verbose_output)
113 {
114  unsigned int rank = dealii::Utilities::MPI::this_mpi_process(communicator);
115  if ((rank == 0) && verbose_output)
116  {
117  std::cout << "Number of unique experimental points is "
118  << std::set<double>(expt_to_dof_mapping.second.begin(),
119  expt_to_dof_mapping.second.end())
120  .size()
121  << std::endl;
122  }
123 
124  auto locally_owned_dofs = temperature.locally_owned_elements();
125  for (unsigned int i = 0; i < points_values.values.size(); ++i)
126  {
127  // Only add locally owned elements
128  if (locally_owned_dofs.is_element(expt_to_dof_mapping.second[i]))
129  temperature[expt_to_dof_mapping.second[i]] = points_values.values[i];
130  }
131 
132  temperature.compress(dealii::VectorOperation::insert);
133 }

135 std::vector<std::vector<double>>
136 read_frame_timestamps(boost::property_tree::ptree const &experiment_database)
137 {
138  // PropertyTreeInput experiment.log_filename
139  std::string log_filename =
140  experiment_database.get<std::string>("log_filename");
141 
142  wait_for_file(log_filename, "Waiting for frame time stamps: " + log_filename);
143 
144  // PropertyTreeInput experiment.first_frame_temporal_offset
145  double first_frame_offset =
146  experiment_database.get("first_frame_temporal_offset", 0.0);
147 
148  // PropertyTreeInput experiment.first_frame
149  unsigned int first_frame =
150  experiment_database.get<unsigned int>("first_frame", 0);
151  // PropertyTreeInput experiment.last_frame
152  unsigned int last_frame = experiment_database.get<unsigned int>("last_frame");
153 
154  // PropertyTreeInput experiment.first_camera_id
155  unsigned int first_camera_id =
156  experiment_database.get<unsigned int>("first_camera_id");
157  // PropertyTreeInput experiment.last_camera_id
158  unsigned int last_camera_id =
159  experiment_database.get<unsigned int>("last_camera_id");
160 
161  unsigned int num_cameras = last_camera_id - first_camera_id + 1;
162  std::vector<std::vector<double>> time_stamps(num_cameras);
163 
164  std::vector<double> first_frame_value(num_cameras);
165 
166  // Read and parse the file
167  std::ifstream file;
168  file.open(log_filename);
169  std::string line;
170  while (std::getline(file, line))
171  {
172  unsigned int entry_index = 0;
173  std::stringstream s_stream(line);
174  bool frame_of_interest = false;
175  unsigned int frame = std::numeric_limits<unsigned int>::max();
176  while (s_stream.good())
177  {
178  std::string substring;
179  std::getline(s_stream, substring, ',');
180  boost::trim(substring);
181 
182  if (entry_index == 0)
183  {
184  std::string error_message = "The file " + log_filename +
185  " does not have consecutive frame indices.";
186  ASSERT_THROW(std::stoi(substring) - frame == 1 ||
187  frame == std::numeric_limits<unsigned int>::max(),
188  error_message.c_str());
189  frame = std::stoi(substring);
190  if (frame >= first_frame && frame <= last_frame)
191  frame_of_interest = true;
192  }
193  else
194  {
195  if (frame == first_frame && substring.size() > 0)
196  first_frame_value[entry_index - 1] = std::stod(substring);
197 
198  if (frame_of_interest && substring.size() > 0)
199  time_stamps[entry_index - 1].push_back(
200  std::stod(substring) - first_frame_value[entry_index - 1] +
201  first_frame_offset);
202  }
203  entry_index++;
204  }
205  }
206  return time_stamps;
207 }
